128 bool insert(const KeyType& value) {
129 std::vector<SkipNode*> update_path(MAX_HEIGHT, nullptr);
130 SkipNode* current = sentinel_head_;
131
132 for (int i = current_height_ - 1; i >= 0; --i) {
133 while (current->forward[i] && current->forward[i]->value < value) {
134 current = current->forward[i];
135 }
136 update_path[i] = current;
137 }
138
139 current = current->forward[0];
140
141 if (current && current->value == value) {
142 return false; // Element already exists
143 }

145 int newHeight = generateRandomHeight();
146 if (newHeight > current_height_) {
147 for (int i = current_height_; i < newHeight; ++i) {
148 update_path[i] = sentinel_head_;
149 }
150 current_height_ = newHeight;

153 SkipNode* newNode = new SkipNode(newHeight, value);
154 for (int i = 0; i < newHeight; ++i) {
155 newNode->forward[i] = update_path[i]->forward[i];
156 update_path[i]->forward[i] = newNode;
157 }
